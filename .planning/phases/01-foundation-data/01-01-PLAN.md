   ```typescript

   import react from '@vitejs/plugin-react';

     plugins: [react(), tailwindcss()],

   ```

3. Replace `src/index.css` content with:
   ```css
   @import "tailwindcss";

   body {
     margin: 0;
     background-color: #1a1a1a;
     color: #fff;
     font-family: 'Verdana', 'Arial', sans-serif;
     overflow: hidden;
   }
   ```
   NOTE: Tailwind v4 uses `@import "tailwindcss"` -- NOT the v3 `@tailwind base/components/utilities` directives. No `tailwind.config.js` file needed.

4. Create `src/types/stock.ts` with shared type definitions:
   ```typescript
   export type Timeframe = 'day' | 'week' | 'month' | 'year';
   export type Exchange = 'all' | 'HOSE' | 'HNX' | 'UPCOM';

   export interface StockData {
     ticker: string;         // e.g., "VNM", "VIC"
     companyName: string;    // e.g., "Vinamilk", "Vingroup"
     exchange: 'HOSE' | 'HNX' | 'UPCOM';
     price: number;          // thousands of VND (e.g., 75.5 = 75,500 VND)
     marketCap: number;      // billions of VND
     changeDay: number;      // % change today
     changeWeek: number;     // % change this week
     changeMonth: number;    // % change this month
     changeYear: number;     // % change this year
   }
   ```

5. Create `src/store/useAppStore.ts` -- Zustand UI state store using the v5 curried pattern:
   ```typescript
   import { create } from 'zustand';
   import type { Timeframe, Exchange } from '../types/stock';

   interface AppState {
     selectedTimeframe: Timeframe;
     selectedExchange: Exchange;
     searchQuery: string;
     setTimeframe: (tf: Timeframe) => void;
     setExchange: (ex: Exchange) => void;
     setSearchQuery: (q: string) => void;
   }

   export const useAppStore = create<AppState>()((set) => ({
     selectedTimeframe: 'day',
     selectedExchange: 'all',
     searchQuery: '',
     setTimeframe: (tf) => set({ selectedTimeframe: tf }),
     setExchange: (ex) => set({ selectedExchange: ex }),
     setSearchQuery: (q) => set({ searchQuery: q }),
   }));
   ```
   IMPORTANT: Use double-parentheses `create<AppState>()((set) => ...)` -- this is required for TypeScript inference in Zustand 5.

6. Update `src/main.tsx` to import index.css and render a minimal App placeholder (just a div with "VN Stock Bubbles" text) to verify the pipeline works. Delete any Vite boilerplate files (App.css, assets/, etc.).

7. Verify `npm run dev` starts without errors and the browser shows the placeholder text on a dark background.

     - Compute radius using sqrt scale: `t = (sqrt(marketCap) - sqrt(minCap)) / (sqrt(maxCap) - sqrt(minCap))`

     - `mass[i] = marketCap` (raw value, physics uses this in Phase 2)

     - vx, vy: leave at 0 (Float32Array default)
